```python
def calculate_fibonacci(n):
    """Calculate the nth Fibonacci number."""
    if n <= 1:
        return n
    return calculate_fibonacci(n-1) + calculate_fibonacci(n-2)

# Example usage
for i in range(10):
    print(f"F({i}) = {calculate_fibonacci(i)}")
```

## MDX Features

### Using Variables

export const greeting = "Hello from MDX!";
export const currentYear = 2026;

{greeting} The current year is {currentYear}.

### Using JSX Elements

<div style={{ padding: '1rem', background: '#f0f0f0', borderRadius: '8px' }}>
  This is a JSX element embedded in MDX
</div>

### Conditional Rendering

export const isLoggedIn = true;

{isLoggedIn && <p>Welcome back! You are logged in.</p>}

export const isAdmin = false;

{isAdmin ? <p>Admin panel access granted.</p> : <p>Regular user access.</p>}

### Mapping Data

export const fruits = ['Apple', 'Banana', 'Cherry', 'Date'];

<ul>
  {fruits.map(fruit => <li key={fruit}>{fruit}</li>)}
</ul>
